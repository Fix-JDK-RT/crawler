            if (!history.pushState) {
                // If the browser doesn't support pushState, we don't want ajax; bail.
                return;
            }
            // Intercept the submit to use ajax
            $(document).on("submit", "#salary-form", function () {
                var url = this.action + "?" + $(this).serialize();
                var moreInfo = $(".salary-note a.salary-more-info");
                moreInfo.removeAttr("href");
                moreInfo.addClass("disabled");
                ajaxUpdate(url, true);
                return false;
            });
            // Handle navigation
            window.onpopstate = function () {
                ajaxUpdate(document.location.href);
            };
            function ajaxUpdate(url, updateHistory) {
                $.get(url, function (html) {
                    // The HTML that comes back is only the div containing the form & salary calc; see controller action for /company/salary/calculator route.
                    // Put the returned markup into a jQuery object, to more easily manipulate it
                    var h = $(html);
                    // Hide that button again!
                    h.find("#salary-form input:submit").hide();
                    
                    // Swap out the div
                    $("#calculator-body").replaceWith(h);
                    // Change the URL
                    if (updateHistory) {
                        history.pushState({}, "", url);
                    }                    
                });
            };
            // üö¢ Shippable here (fast ajax update and back button works!)
        })();
